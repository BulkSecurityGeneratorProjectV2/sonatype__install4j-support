13  package org.sonatype.install4j.maven;
14  
15  import org.apache.maven.plugins.annotations.Mojo;
16  import org.apache.maven.plugins.annotations.Parameter;
17  import org.apache.tools.ant.taskdefs.ExecTask;
18  
19  /**
20   * Update the install4j license key (via install4jc --license).
21   *
22   * Execution will skip if <code>licenseKey</code> parameter is not configured, or the <code>install4j.home</code> location is invalid.
23   *
24   * @since 1.0
25   */
26  @Mojo(name="install-license")
27  public class InstallLicenseMojo
28      extends Install4jcMojoSupport
29  {
30      /**
31       * install4j license key.
32       */
33      @Parameter(property="install4j.licenseKey")
34      private String licenseKey;
35  
36      @Override
37      protected void execute(final AntHelper ant, final ExecTask task) throws Exception {
38          if (licenseKey == null) {
39              log.warn("Missing install4j.licenseKey; skipping");
40              return;
41          }
42          task.createArg().setValue("--license");
43          task.createArg().setValue(licenseKey);
44          task.execute();
45      }
46  }
