14  package org.sonatype.install4j.maven;
15  
16  import org.apache.maven.plugins.annotations.Component;
17  import org.apache.maven.plugins.annotations.Parameter;
18  import org.apache.maven.project.MavenProject;
19  import org.apache.tools.ant.taskdefs.ExecTask;
20  import org.codehaus.plexus.util.Os;
21  
22  import java.io.File;
23  
24  /**
25   * Support for install4jc-based tasks.
26   *
27   * Compatible with install4j version 5.1.2 or higher.
28   *
29   * @since 1.0
30   */
31  public abstract class Install4jcMojoSupport
32      extends MojoSupport
33  {
34      /**
35       * Skip execution.
36       */
37      @Parameter(property = "install4j.skip", defaultValue = "false")
38      protected boolean skip;
39  
40      /**
41       * The location of the install4j installation.
42       */
43      @Parameter(property = "install4j.home", required = true)
44      protected File installDir;
45  
46      @Component
47      protected MavenProject project;
48  
49      @Override
50      protected void doExecute() throws Exception {
51          if (skip) {
52              log.warn("Skipping execution");
53              return;
54          }
55  
56          AntHelper ant = new AntHelper(this, project);
57  
58          if (!installDir.exists()) {
59              log.warn("Invalid install directory; skipping: " + installDir);
60              return;
61          }
62  
63          log.debug("install4j installation directory: " + installDir);
64  
65          boolean windows = Os.isFamily(Os.FAMILY_WINDOWS);
66          File install4jc = new File(installDir, "bin/install4jc" + (windows ? ".exe" : ""));
67  
68          if (!install4jc.exists()) {
69              log.warn("Missing install4j compiler executable: " + install4jc);
70              return;
71          }
72  
73          log.debug("install4jc: " + install4jc);
74  
75          // ensure the binary is executable
76          ant.chmod(install4jc, "u+x");
77  
78          // ensure version is compatible
79          VersionHelper versionHelper = new VersionHelper(log);
80          String versionProperty = versionHelper.fetchVersion(ant, install4jc);
81          versionHelper.ensureVersionCompatible(ant.getProperty(versionProperty));
82  
83          ExecTask task = ant.createTask(ExecTask.class);
84          task.setExecutable(install4jc.getAbsolutePath());
85          execute(ant, task);
86      }
87  
88      protected abstract void execute(final AntHelper ant, final ExecTask task) throws Exception;
89  }
