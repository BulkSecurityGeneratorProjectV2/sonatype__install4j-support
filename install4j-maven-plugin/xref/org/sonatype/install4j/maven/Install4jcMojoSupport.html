16  import org.apache.maven.plugin.MojoExecutionException;
17  import org.apache.maven.plugins.annotations.Component;
18  import org.apache.maven.plugins.annotations.Parameter;
19  import org.apache.maven.project.MavenProject;

21  import org.codehaus.plexus.util.Os;
22  import org.sonatype.aether.util.version.GenericVersionScheme;
23  import org.sonatype.aether.version.Version;
24  import org.sonatype.aether.version.VersionConstraint;
25  import org.sonatype.aether.version.VersionScheme;

27  import java.io.File;

29  /**
30   * Support for install4jc-based tasks.
31   *
32   * Compatible with install4j version 5.1.2 or higher.
33   *
34   * @since 1.0
35   */
36  public abstract class Install4jcMojoSupport
37      extends MojoSupport
38  {
39      private static final String VERSION_CONSTRAINT = "[5.1.2,)"; // allow 5.1.2+
40  
41      /**
42       * Skip execution.
43       */
44      @Parameter(property = "install4j.skip", defaultValue = "false")
45      protected boolean skip;
46  
47      /**
48       * The location of the install4j installation.
49       */
50      @Parameter(property = "install4j.home", required = true)
51      protected File installDir;
52  
53      @Component
54      protected MavenProject project;
55  
56      @Override
57      protected void doExecute() throws Exception {
58          if (skip) {
59              log.warn("Skipping execution");
60              return;
61          }
62  
63          AntHelper ant = new AntHelper(this, project);
64  
65          if (!installDir.exists()) {
66              log.warn("Invalid install directory; skipping: " + installDir);
67              return;
68          }
69  
70          log.debug("install4j installation directory: " + installDir);
71  
72          boolean windows = Os.isFamily(Os.FAMILY_WINDOWS);
73          File install4jc = new File(installDir, "bin/install4jc" + (windows ? ".exe" : ""));
74  
75          if (!install4jc.exists()) {
76              log.warn("Missing install4j compiler executable: " + install4jc);
77              return;
78          }
79  
80          log.debug("install4jc: " + install4jc);
81  
82          ant.chmod(install4jc, "u+x");
83  
84          // Sanity check, ask install4jc for its version
85          ExecTask task = ant.createTask(ExecTask.class);
86          task.setExecutable(install4jc.getAbsolutePath());
87          task.createArg().setValue("--version");
88          // ensure we have a fresh property to return the version details in
89          String versionProperty = "install4j.version-" + System.currentTimeMillis();
90          task.setOutputproperty(versionProperty);
91          task.execute();
92          ensureVersionCompatible(ant.getProperty(versionProperty));
93  
94          task = ant.createTask(ExecTask.class);
95          task.setExecutable(install4jc.getAbsolutePath());
96          execute(ant, task);
97      }
98  
99      // TODO: Convert helpers into components so we can test them, have to sort out slf4j maven integration

101     /**
102      * Parse version in format:
103      *
104      * install4j version _version-#_ (build _build-#_), built on _date_
105      */
106     private String parseVersion(final String rawVersion) {
107         log.debug("Parsing version: " + rawVersion);
108         String[] parts = rawVersion.split("\\s");
109         boolean valid = parts.length > 3 && parts[0].equals("install4j") && parts[1].equals("version");
110         if (!valid) {
111             throw new RuntimeException("Unable to parse version from input: " + rawVersion);
112         }
113         // ignore the build #
114         return parts[2];
115     }
116 
117     private void ensureVersionCompatible(final String rawVersion) throws Exception {
118         String version = parseVersion(rawVersion);
119         VersionScheme scheme = new GenericVersionScheme();
120         VersionConstraint constraint = scheme.parseVersionConstraint(VERSION_CONSTRAINT);
121         Version _version = scheme.parseVersion(version);
122         log.debug("Version: " + _version);
123 
124         if (!constraint.containsVersion(_version)) {
125             log.error("Incompatible install4j version detected");
126             log.error("Raw version: " + rawVersion);
127             log.error("Detected version: " + _version);
128             log.error("Compatible version constraint: " + constraint);
129             throw new MojoExecutionException("Unsupported install4j version: " + rawVersion);
130         }
131     }
132 
133     protected abstract void execute(final AntHelper ant, final ExecTask task) throws Exception;
134 }
